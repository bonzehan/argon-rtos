217  ar_status_t init(const char * name, T * object, void (T::*entry)(), void * stack, unsigned stackSize, uint8_t priority)
218  {
219  // Invoke the base initializer, passing the object pointer as the entry param.
220  ar_status_t result = init(name, member_thread_entry<T>, static_cast<void *>(object), stack, stackSize, priority);
221  if (result == kArSuccess)
222  {
223  // Save the member function pointer on the thread's stack.
224  uint8_t * storage = m_stackTop - (sizeof(ThreadContext) + sizeof(entry));
225  memcpy(storage, &entry, sizeof(entry));
226  }
227  return result;
228  }
